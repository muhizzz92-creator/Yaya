import torch import torch.nn as nn import os import sys import argparse import torchaudio from safetensors.torch import load_file from tortoise.api import TextToSpeech # Настройка путей (относительно скрипта) CURRENT_DIR = os.path.dirname(os.path.abspath(__file__)) # Путь к папке с вашей моделью Chatterbox MODEL_DIR = os.path.join(CURRENT_DIR, "models", "Chatterbox") MODEL_PATH = os.path.join(MODEL_DIR, "t3_turbo_v1.safetensors") def generate_audio(text, output_path): # 1. Проверка наличия модели if not os.path.exists(MODEL_PATH): print(f"ERROR: Model file not found at {MODEL_PATH}", file=sys.stderr) return False try: # 2. Инициализация движка (без авто-загрузки весов) # use_deepspeed=False для стабильности на Windows tts = TextToSpeech(use_deepspeed=False, kv_cache=True) # 3. ХАК АРХИТЕКТУРЫ (Критически важно для Chatterbox!) # Стандартная модель имеет размер 256. Chatterbox - 50276. # Мы вручную расширяем слои перед загрузкой весов. tts.autoregressive.text_token_embedding = nn.Embedding(50276, 1024) tts.autoregressive.text_head = nn.Linear(1024, 50276, bias=False) # 4. Загрузка ваших весов print(f"Loading weights from {MODEL_PATH}...") ar_weights = load_file(MODEL_PATH) tts.autoregressive.load_state_dict(ar_weights, strict=False) # 5. Генерация # preset="fast" или "ultra_fast" для скорости # voice_samples=None, conditioning_latents=None - используем рандомный голос, # так как у нас нет папки с голосами (voices). # Если нужно конкретный голос - нужно добавить папку voices/ print(f"Generating: {text[:30]}...") # Генерируем с рандомным голосом (пока так, чтобы заработало) gen = tts.tts_with_preset(text, preset="fast") # 6. Сохранение # gen - это тензор (1, samples). Squeeze убирает лишнюю размерность. torchaudio.save(output_path, gen.squeeze(0).cpu(), 24000) print(f"SUCCESS: {output_path}") return True except Exception as e: # Выводим полный текст ошибки в stderr, чтобы программа его поймала import traceback traceback.print_exc() print(f"CRITICAL ERROR: {e}", file=sys.stderr) return False if __name__ == "__main__": parser = argparse.ArgumentParser() parser.add_argument("text", help="Text to speak") parser.add_argument("--filename", required=True, help="Output file path") args = parser.parse_args() generate_audio(args.text, args.filename)
